## 二、扫描功能模块

### 2.1 实时相机扫描

#### 核心功能

```kotlin
/**
 * 扫描配置
 */
data class ScanConfig(
    // 扫描模式
    val scanMode: ScanMode = ScanMode.AUTO,

    // 支持的码类型
    val supportedFormats: List<BarcodeFormat> = BarcodeFormat.entries,

    // 自动检测
    val autoDetect: Boolean = true,

    // 自动对焦
    val autoFocus: Boolean = true,

    // 扫描区域（ROI）
    val scanRegion: Rect? = null,

    // 连续扫描
    val continuousScan: Boolean = false,

    // 震动反馈
    val hapticFeedback: Boolean = true,

    // 声音提示
    val soundEnabled: Boolean = true,

    // 自动跳转
    val autoOpenUrl: Boolean = false,

    // 安全检查
    val securityCheck: Boolean = true
)

enum class ScanMode {
    AUTO,           // 自动检测所有类型
    QR_ONLY,        // 仅 QR 码
    BARCODE_ONLY,   // 仅条码
    CUSTOM          // 自定义
}
```

**2.5.1 智能解析**

```kotlin
/**
 * 扫描结果智能解析
 */
sealed class ScanResult {
    data class Url(
        val url: String,
        val title: String? = null,
        val description: String? = null,
        val favicon: ByteArray? = null,
        val securityLevel: SecurityLevel
    ) : ScanResult()

    data class Text(
        val text: String,
        val language: String? = null,
        val keywords: List<String>? = null
    ) : ScanResult()

    data class Contact(
        val name: String,
        val phone: String?,
        val email: String?,
        val address: String?,
        val organization: String?,
        val vCard: String
    ) : ScanResult()

    data class Wifi(
        val ssid: String,
        val password: String,
        val encryption: WifiEncryption,
        val isHidden: Boolean
    ) : ScanResult()

    data class Email(
        val to: String,
        val subject: String?,
        val body: String?,
        val cc: List<String>?,
        val bcc: List<String>?
    ) : ScanResult()

    data class Sms(
        val phone: String,
        val message: String
    ) : ScanResult()

    data class Phone(
        val number: String,
        val type: PhoneType
    ) : ScanResult()

    data class Calendar(
        val title: String,
        val startTime: Long,
        val endTime: Long?,
        val location: String?,
        val description: String?,
        val attendees: List<String>?
    ) : ScanResult()

    data class Location(
        val latitude: Double,
        val longitude: Double,
        val altitude: Double?,
        val name: String?
    ) : ScanResult()

    data class Product(
        val barcode: String,
        val name: String?,
        val price: Double?,
        val currency: String?,
        val imageUrl: String?,
        val productUrl: String?
    ) : ScanResult()

    data class ISBN(
        val isbn: String,
        val title: String?,
        val author: String?,
        val publishYear: Int?,
        val coverUrl: String?
    ) : ScanResult()

    data class Unknown(
        val rawValue: String,
        val format: BarcodeFormat,
        val metadata: Map<String, Any>
    ) : ScanResult()
}

enum class SecurityLevel {
    SAFE,           // 安全
    SUSPICIOUS,     // 可疑
    UNSAFE,         // 不安全
    UNKNOWN         // 未知
}
```
